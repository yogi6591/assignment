#GIT Commands

#GIT Configurations

git config --global user.name "Your full name"
git config --global user.email "Your email"


#Starting a repository

git init
git status #for checking the changes made after or before updation


# Stagging files

git add filename #for selective approach
git add . #for all files


#Committing to a repository

git commit -m"Your message"
git log. #shows all the commits that we made


#Pulling and pushing from and to repository

git remote add origin #Copy the link from github repository
git push origin master # Tap enter and wait for compiling


#GIT Updations

git add .
git commit -m"Your message"
git status
git push origin master
